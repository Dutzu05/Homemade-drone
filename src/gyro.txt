#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>
#include <Wire.h>

Adafruit_MPU6050 mpu;

// --- CALIBRATION OFFSETS ---
// These are subtracted from the raw readings to center them at 0.00
float gyroX_offset = -0.07;
float gyroY_offset = -0.02;
float gyroZ_offset = 0.00;

void setup(void) {
  Serial.begin(115200);
  while (!Serial) delay(10); 

  Wire.begin(8, 9); 

  Serial.println("Initializing GY-521 with Calibration...");

  if (!mpu.begin()) {
    Serial.println("Failed to find MPU6050 chip!");
    while (1) { delay(10); }
  }

  mpu.setGyroRange(MPU6050_RANGE_250_DEG);
  mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);

  Serial.println("Calibration Applied. Readings should now be near 0.00 at rest.");
  delay(100);
}

void loop() {
  sensors_event_t a, g, temp;
  mpu.getEvent(&a, &g, &temp);

  // Apply calibration: Corrected Value = Raw Value - Offset
  float calibrated_X = g.gyro.x - gyroX_offset;
  float calibrated_Y = g.gyro.y - gyroY_offset;
  float calibrated_Z = g.gyro.z - gyroZ_offset;

  /* Output Calibrated Data */
  Serial.print("G_X: "); Serial.print(calibrated_X);
  Serial.print(" | G_Y: "); Serial.print(calibrated_Y);
  Serial.print(" | G_Z: "); Serial.println(calibrated_Z);

  delay(500); 
}
