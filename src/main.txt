#include <Arduino.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>
#include <Wire.h>

Adafruit_MPU6050 mpu;
float roll = 0.0f, pitch = 0.0f, yaw = 0.0f;
float rollTrim = 0.0f, pitchTrim = 0.0f;
float gyroX_offset = -0.07f;
float gyroY_offset = -0.02f;
float gyroZ_offset = 0.00f;
unsigned long lastTime;
unsigned long lastTelemetryTime;
unsigned long lastPotChangeTime;
const unsigned long TELEMETRY_INTERVAL_MS = 100;

const int POT_PIN = 0;

const int PWM_PINS[] = {3, 4, 6, 7};
const int NUM_PWM_PINS = 4;

const int PWM_CHANNELS[] = {0, 1, 2, 3};

const int MOTOR_REPORT_PINS[] = {4, 3, 7, 6};
const char* MOTOR_NAMES[] = {"Motor 1", "Motor 2", "Motor 3", "Motor 4"};

const int PWM_FREQ = 20000;
const int PWM_RESOLUTION = 8;
const int PWM_MAX = 255;

const int ADC_RESOLUTION = 12;
const int ADC_MAX = 4095;

const float BATTERY_VOLTAGE = 3.6f;
const float MOTOR_RESISTANCE = 0.5f;
const float DEG_PER_RAD = 57.2958f;
const float STILL_GYRO_THRESHOLD = 0.03f;
const float STILL_ACCEL_TOLERANCE = 0.6f;
const float ANGLE_DEADBAND_DEG = 2.0f;
const float ROLL_KP = 2.8f;
const float ROLL_KD = 0.35f;
const float PITCH_KP = 2.8f;
const float PITCH_KD = 0.35f;
const int MAX_AXIS_CORRECTION = 45;
const int POT_STABILITY_THRESHOLD = 12;
const unsigned long POT_STABLE_TIME_MS = 250;
const int MIN_STABILIZE_PWM = 35;
const int POT_FILTER_SAMPLES = 8;

int lastPotValue = 0;
int filteredPotValue = 0;

float estimateAverageVoltage(int dutyCycle) {
  return BATTERY_VOLTAGE * ((float)dutyCycle / (float)PWM_MAX);
}

float estimateCurrent(float voltage) {
  if (voltage < 0.1f) return 0.0f;
  return voltage / MOTOR_RESISTANCE;
}

int readPotAveraged() {
  long sum = 0;
  for (int i = 0; i < POT_FILTER_SAMPLES; i++) {
    sum += analogRead(POT_PIN);
    delayMicroseconds(150);
  }
  return (int)(sum / POT_FILTER_SAMPLES);
}

int getPwmChannelForPin(int pin) {
  for (int i = 0; i < NUM_PWM_PINS; i++) {
    if (PWM_PINS[i] == pin) {
      return PWM_CHANNELS[i];
    }
  }
  return -1;
}

void printMotorTelemetry(const char* motorName, int pin, int dutyCycle) {
  float vAvg = estimateAverageVoltage(dutyCycle);
  float iEst = estimateCurrent(vAvg);

  Serial.print(motorName);
  Serial.print(" | pin=");
  Serial.print(pin);
  Serial.print(" | pwm=");
  Serial.print(dutyCycle);
  Serial.print(" | v=");
  Serial.print(vAvg, 2);
  Serial.print("V | curr=");
  Serial.print(iEst, 2);
  Serial.print("A | rpm=N/A");
  Serial.println();
}

float applyDeadband(float value, float deadband) {
  if (fabs(value) <= deadband) {
    return 0.0f;
  }
  return value > 0.0f ? value - deadband : value + deadband;
}

void computeMotorOutputs(int baseDutyCycle,
                         int potValue,
                         float rollErrorDeg,
                         float pitchErrorDeg,
                         float gyroX,
                         float gyroY,
                         int motorDutyCycles[NUM_PWM_PINS],
                         float &rollCorrection,
                         float &pitchCorrection,
                         bool &stabilizationActive) {
  for (int i = 0; i < NUM_PWM_PINS; i++) {
    motorDutyCycles[i] = baseDutyCycle;
  }

  if (abs(potValue - lastPotValue) > POT_STABILITY_THRESHOLD) {
    lastPotChangeTime = millis();
  }
  lastPotValue = potValue;

  bool potStable = (millis() - lastPotChangeTime) >= POT_STABLE_TIME_MS;
  stabilizationActive = potStable && baseDutyCycle >= MIN_STABILIZE_PWM;

  if (!stabilizationActive) {
    rollCorrection = 0.0f;
    pitchCorrection = 0.0f;
    return;
  }

  float rollRateDeg = gyroX * DEG_PER_RAD;
  float pitchRateDeg = gyroY * DEG_PER_RAD;

  float rollErrorForControl = applyDeadband(rollErrorDeg, ANGLE_DEADBAND_DEG);
  float pitchErrorForControl = applyDeadband(pitchErrorDeg, ANGLE_DEADBAND_DEG);

  rollCorrection = -(ROLL_KP * rollErrorForControl + ROLL_KD * rollRateDeg);
  pitchCorrection = -(PITCH_KP * pitchErrorForControl + PITCH_KD * pitchRateDeg);

  rollCorrection = constrain(rollCorrection, -MAX_AXIS_CORRECTION, MAX_AXIS_CORRECTION);
  pitchCorrection = constrain(pitchCorrection, -MAX_AXIS_CORRECTION, MAX_AXIS_CORRECTION);

  float motorAdjustments[NUM_PWM_PINS];
  motorAdjustments[0] = rollCorrection + pitchCorrection;   // Motor 1 -> pin 4
  motorAdjustments[1] = rollCorrection - pitchCorrection;   // Motor 2 -> pin 3
  motorAdjustments[2] = -rollCorrection - pitchCorrection;  // Motor 3 -> pin 7
  motorAdjustments[3] = -rollCorrection + pitchCorrection;  // Motor 4 -> pin 6

  for (int i = 0; i < NUM_PWM_PINS; i++) {
    int adjustedDuty = baseDutyCycle + (int)round(motorAdjustments[i]);
    motorDutyCycles[i] = constrain(adjustedDuty, 0, PWM_MAX);
  }
}

void updateImuTelemetry(float &accX, float &accY, float &accZ,
                        float &gyroX, float &gyroY, float &gyroZ) {
  sensors_event_t a, g, temp;
  mpu.getEvent(&a, &g, &temp);

  unsigned long currentTime = micros();
  float dt = (currentTime - lastTime) / 1000000.0f;
  lastTime = currentTime;

  accX = a.acceleration.x;
  accY = a.acceleration.y;
  accZ = a.acceleration.z;

  gyroX = g.gyro.x - gyroX_offset;
  gyroY = g.gyro.y - gyroY_offset;
  gyroZ = g.gyro.z - gyroZ_offset;

  float accRoll = atan2(accY, accZ) * DEG_PER_RAD;
  float accPitch = atan2(-accX, sqrt(accY * accY + accZ * accZ)) * DEG_PER_RAD;
  float accelMagnitude = sqrt(accX * accX + accY * accY + accZ * accZ);
  bool isStill =
    fabs(gyroX) < STILL_GYRO_THRESHOLD &&
    fabs(gyroY) < STILL_GYRO_THRESHOLD &&
    fabs(gyroZ) < STILL_GYRO_THRESHOLD &&
    fabs(accelMagnitude - 9.81f) < STILL_ACCEL_TOLERANCE;

  float gyroWeight = isStill ? 0.90f : 0.96f;
  float accelWeight = 1.0f - gyroWeight;

  roll = gyroWeight * (roll + gyroX * DEG_PER_RAD * dt) + accelWeight * accRoll;
  pitch = gyroWeight * (pitch + gyroY * DEG_PER_RAD * dt) + accelWeight * accPitch;
  yaw = yaw + gyroZ * DEG_PER_RAD * dt;
}

void calibrateLevelTrim() {
  const int samples = 200;
  float accRollSum = 0.0f;
  float accPitchSum = 0.0f;

  for (int i = 0; i < samples; i++) {
    sensors_event_t a, g, temp;
    mpu.getEvent(&a, &g, &temp);

    float accRoll = atan2(a.acceleration.y, a.acceleration.z) * 57.2958f;
    float accPitch = atan2(
      -a.acceleration.x,
      sqrt(a.acceleration.y * a.acceleration.y + a.acceleration.z * a.acceleration.z)
    ) * 57.2958f;

    accRollSum += accRoll;
    accPitchSum += accPitch;
    delay(5);
  }

  rollTrim = accRollSum / samples;
  pitchTrim = accPitchSum / samples;
  roll = 0.0f;
  pitch = 0.0f;
  yaw = 0.0f;
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  Wire.begin(8, 9);

  if (!mpu.begin()) {
    Serial.println("MPU6050 connection failed!");
  } else {
    mpu.setAccelerometerRange(MPU6050_RANGE_8_G);
    mpu.setGyroRange(MPU6050_RANGE_250_DEG);
    mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);
    Serial.println("Keep drone still: calibrating level trim...");
    calibrateLevelTrim();
  }

  analogReadResolution(ADC_RESOLUTION);

  filteredPotValue = readPotAveraged();
  lastPotValue = filteredPotValue;

  for (int i = 0; i < NUM_PWM_PINS; i++) {
    ledcSetup(PWM_CHANNELS[i], PWM_FREQ, PWM_RESOLUTION);
    ledcAttachPin(PWM_PINS[i], PWM_CHANNELS[i]);
    ledcWrite(PWM_CHANNELS[i], 0);
  }

  lastTime = micros();
  lastTelemetryTime = millis();
  lastPotChangeTime = millis();
}

void loop() {
  float accX = 0.0f;
  float accY = 0.0f;
  float accZ = 0.0f;
  float gyroX = 0.0f;
  float gyroY = 0.0f;
  float gyroZ = 0.0f;
  updateImuTelemetry(accX, accY, accZ, gyroX, gyroY, gyroZ);

  int rawPotValue = readPotAveraged();
  filteredPotValue = (filteredPotValue * 3 + rawPotValue) / 4;

  int baseDutyCycle = map(filteredPotValue, 0, ADC_MAX, 0, PWM_MAX);
  baseDutyCycle = constrain(baseDutyCycle, 0, PWM_MAX);

  float rollErrorDeg = roll - rollTrim;
  float pitchErrorDeg = pitch - pitchTrim;
  int motorDutyCycles[NUM_PWM_PINS];
  float rollCorrection = 0.0f;
  float pitchCorrection = 0.0f;
  bool stabilizationActive = false;

  computeMotorOutputs(baseDutyCycle, filteredPotValue, rollErrorDeg, pitchErrorDeg,
                      gyroX, gyroY, motorDutyCycles, rollCorrection,
                      pitchCorrection, stabilizationActive);

  for (int i = 0; i < NUM_PWM_PINS; i++) {
    int channel = getPwmChannelForPin(MOTOR_REPORT_PINS[i]);
    if (channel >= 0) {
      ledcWrite(channel, motorDutyCycles[i]);
    }
  }

  if (millis() - lastTelemetryTime >= TELEMETRY_INTERVAL_MS) {
    lastTelemetryTime = millis();

    Serial.println("----- Telemetry Snapshot -----");

    Serial.print("POSITION | Ruliu: ");
    Serial.print(rollErrorDeg, 1);
    Serial.print(" deg | Tangaj: ");
    Serial.print(pitchErrorDeg, 1);
    Serial.print(" deg | Girație (Yaw): ");
    Serial.print(yaw, 1);
    Serial.println(" deg");

    Serial.print("ACCEL    | X: ");
    Serial.print(accX, 2);
    Serial.print(" | Y: ");
    Serial.print(accY, 2);
    Serial.print(" | Z: ");
    Serial.print(accZ, 2);
    Serial.println(" m/s^2");

    Serial.print("GYRO     | X: ");
    Serial.print(gyroX, 3);
    Serial.print(" | Y: ");
    Serial.print(gyroY, 3);
    Serial.print(" | Z: ");
    Serial.print(gyroZ, 3);
    Serial.println(" rad/s");

    Serial.print("INPUT    | Pot: ");
    Serial.print(rawPotValue);
    Serial.print(" | PotFilt: ");
    Serial.print(filteredPotValue);
    Serial.print(" | PWM_CMD: ");
    Serial.println(baseDutyCycle);

    Serial.print("CTRL     | mode=");
    Serial.print(stabilizationActive ? "HOLD" : "THROTTLE");
    Serial.print(" | roll_corr=");
    Serial.print(rollCorrection, 1);
    Serial.print(" | pitch_corr=");
    Serial.println(pitchCorrection, 1);

    for (int i = 0; i < NUM_PWM_PINS; i++) {
      printMotorTelemetry(MOTOR_NAMES[i], MOTOR_REPORT_PINS[i], motorDutyCycles[i]);
    }

    Serial.println();
  }

  delay(10);
}
